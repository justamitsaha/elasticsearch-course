    "bool": {
      "filter": [
        {
          "range": {
            "price": {

# AND condition - score is 0
POST /products/_search
{
  "query": {
    "bool": {
      "filter": [
        {
          "range": {
            "price": {
              "lte": 100
            }
          }
        },
        {
          "range": {
            "rating": {
              "gte": 4.5
            }

        }
      ]
    }
  }
}

# Let's exclude Nike brand in the above query

POST /products/_search
{
  "query": {
    "bool": {
      "filter": [
        {
          "range": {
            "price": {
              "lte": 100
            }
          }
        },
        {
          "range": {
            "rating": {
              "gte": 4.5
            }
          }
        }
      ],
      "must_not": [
        {
            "term": {
              "brand": {
                "value": "Nike"
              }
            }
        }
      ]
    }
  }
}

# filter vs must
POST /products/_search
{
  "query": {
    "bool": {
      "must": [
        {
          "range": {
            "price": {
              "lte": 100
            }
          }
        }
      ]
    }
  }
}

# better example of must
POST /products/_search
{
  "query": {
    "bool": {
      "filter": [
        {
          "range": {
            "price": {
              "lte": 100
            }
          }
        }
      ],
      "must": [
        {
            "match": {
              "name": "running shoe"
            }
        }
      ]
    }
  }
}
