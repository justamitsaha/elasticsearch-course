{
  "query": {
    "multi_match": {
      "query": "spring season",
      "fields": [ "title", "body" ],
      "operator": "and"
    }
  },
  "highlight": {
    "fields": {
      "title": {
        "pre_tags": [ "<b><i>" ],
        "post_tags": [ "</i></b>" ]

# we can move the common tags
POST /articles/_search
{
  "query": {
    "multi_match": {
      "query": "spring season",
      "fields": [ "title", "body" ],
      "operator": "and"
    }
  },
  "highlight": {
    "pre_tags": [ "<b>" ],
    "post_tags": [ "</b>" ],
    "fields": {
      "body": {},
      "title": {}
    }
  }
}
